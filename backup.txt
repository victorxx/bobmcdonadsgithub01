        // Função para gerar conteúdo randomizado
        function generateRandomContent() {
            // Gerando número aleatório entre 1 e 100
            const randomNumber = Math.floor(Math.random() * 100) + 1;

            // Gerando uma letra aleatória
            const letters = 'ABCDEFGHIJKLMNOPQRSTUVWXYZ';
            const randomLetter = letters.charAt(Math.floor(Math.random() * letters.length));

            // Exibindo o conteúdo randomizado
            const randomContent = `Número aleatório: ${randomNumber} | Letra aleatória: ${randomLetter}`;
            document.getElementById('random-content').textContent = randomContent;
        }
